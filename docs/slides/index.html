Write your tests using Jasmine and WebdriverJS

describe('by model', function() {
   it('should find an element by text input model', function() {
     var username = element(by.model('username'));
     var name = element(by.binding('username'));

     username.clear();
     expect(name.getText()).toEqual('');

     username.sendKeys('Cristian Bancu');
     expect(name.getText()).toEqual('Cristian Bancu');
   });
});

Searching for elements on the page

element() vs element.all()

Single element

element(by.binding('appName') );
element(by.css('.mail-setting-div'));

// HINT
// when selecting a single element by css, a shortcut is available
// the above statement is equivalent to 
$('.mail-setting-div')

Collection of elements

// clicks the 3rd element
element.all( by.css('[ng-click="openPage()"]') ).get(2).click();

// HINT
// when selecting by css, some shorcuts are available 
// the above is equivalent to 
$$('[ng-click="openPage()"]').get(2).click();

ElementFinder


var el = element(locator);

// Click on the element
el.click();

// Send keys to the element (usually an input)
el.sendKeys('my text');

// Clear the text in an element (usually an input)
el.clear();

// Get the value of an attribute, for example, get the value of an input
el.getAttribute('value');	


// Number of elements.
element.all(locator).count();

// Get by index (starting at 0).
element.all(locator).get(index);

// First and last.
element.all(locator).first();

by.binding

In your test

element( by.binding('myModel') );

In your application

<span ng-bind="myModel"></span>
<!-- or -->
<span>{{myModel}}</span>

by.model

In your test

element( by.model('myModel') );

In your application

<input ng-model="myModel" />

by.repeater

In your test


// select element  user.city, on 1st row 
// @returns {ElementFinder}
element( by.repeater('user in users').row(0).column('city') );

// select all element on 3rd row
// remark that element.all need to be used (collection of elements)
// @returns {ElementFinder|*}
element.all(by.repeater('user in users').row(2)); // -or- return all rows/columns with element.all(by.repeater('user in users'))

In your application

<ul>
  <li ng-repeat="user in users">
      <span>{{user.name}}</span>
      <span>{{user.city}}</span>
      <span>{{user.address}}</span>
  </li>
</ul>

by.css

In your test

element( by.css('[ng-click="sendMail()"]') );

In your application

<button ng-click="sendMail()">Send mail!</button>

Promises and the Control Flow

Promises based(1)

All Protractor methods are asynchronous and return promises.

// Example of getText() promise
this.getTemplatesCountDisplayed = function() {
	// e.g. '54 templates found'
	return self.getRecentTemplatesCountElement().getText()
	    .then(function(value) {
	        return parseInt(value.split(" ")[0]);
	    });
	};

Promises with element chaining


this.getGridColumnsNumber = function() {
        return gridTestUtils.getGrid(GRID_ID)
            .element(by.css('.ui-grid-render-container-body'))
            .element(by.css('.ui-grid-header-cell-wrapper'))
            .all(by.repeater('col in colContainer.renderedColumns'))
            .then(function(cols) {
                return cols.length;
            });
    };

Control Flow

WebDriverJS maintains a queue of pending promises, called the control flow, to keep execution organized.


it('should delete the newly created template', function () {
        var gridId = searchTemplatesPage.getGridId();

        expect(recentTemplatesPage.getTemplatesCountDisplayed()).toEqual(1);

        // select 1st the row 
        gridUtils.selectRow(gridId, 0);

        // press delete all on page 
        recentTemplatesPage.deleteAllBtn.click();

        // press delete on modal (for template confirmation)
        recentTemplatesPage.deleteTemplateBtn.click();

        // let some time to refresh the child page (and grid to re-populate)
        browser.driver.sleep(5000);

        // after delete, no template shoud be found in grid 
        expect(recentTemplatesPage.getTemplatesCountDisplayed()).toEqual(0);
    });

In the example above, the control flow would execute the queue following the sequence we see in code. Basically method click() would only run once browser.get promise is resolved, and so on.

Best practices to organize your tests

The big picture

  • Page Objects - These are the js files where you map the elements and write the functions to perform actions;
  • Exports and Require - This is how you connect your Page Objects to your Test Specs;
  • Test specs - These are the js files where you write your tests using jasmine syntax.

Page Objects



var colUtils = require('../test-utils/element-collections-utils');

var SearchTemplatesPage = function() {
    var GRID_ID = "tmplGridId";
    this.id = element(by.model('selectedId')); 
    this.content = element(by.model('selectedContent'));
    this.categoriesComboBox = element(by.model('selectedCategories'));

    this.searchBtn = $('[ng-click="onSearch()"]');
    this.clearBtn = $('[ng-click="onCancel()"]');

    this.get = function () {
        var url = 'https://localhost/MailerAdmin/#/gallery/search/search/search-results?shouldLoadData=false&isSearch=false';
        browser.get(url);
    };

    this.getGridId = function () {
        return GRID_ID;
    };

    this.searchBy = function(id, content, categories) {
        this.clearBtn.click(); // research search criteria 
        if (id != null && !isNaN(id)) {
            this.id.clear();
            this.id.sendKeys(id);
        }
        if (content != null) {
            this.content.clear();
            this.content.sendKeys(content);
        }
        if (categories != null && categories.length > 0) {
            colUtils.selectComboBoxValues(this.categoriesComboBox, categories);
        }
        this.searchBtn.click();
    };
};
module.exports = new SearchTemplatesPage();

Separate your tests in various test suites


exports.config = {
    // The address of a running selenium server.
    seleniumAddress: 'http://localhost:4444/wd/hub',

    // Spec patterns are relative to the location of the spec file. They may
    // include glob patterns.
    suites: {
        toolbar: 'e2e/toolbar/**/*Spec.js',
        mail_settings: 'e2e/mail-settings/**/*Spec.js',
        recent_templates: 'e2e/recent-templates/**/*Spec.js',
        cross_page : 'e2e/cross-page/**/Spec.js'
    },

    capabilities: {
        'browserName': 'chrome'
    },

    // Options to be passed to Jasmine-node.
    jasmineNodeOpts: {
        showColors: true, // Use colors in the command line report.
        isVerbose: false,
        //includeStackTrace: true
    }

protractor conf.js --suite cross_page

Using onPrepare

Export xml results of your Automated Suites

First, install jasmine-reporters:

npm install jasmine-reporters

And to keep xml results in Timestamp directories, install mkdirp package:

npm install mkdirp

Using onPrepare

Export xml results of your Automated Suites

//config.js
exports.config = {
  onPrepare: function() {
    var folderName = (new Date()).toString().split(' ').splice(1, 4).join(' ');
    var mkdirp = require('mkdirp');
    var newFolder = "./reports/" + folderName;
    require('jasmine-reporters');

    mkdirp(newFolder, function(err) {
      if (err) {
        console.error(err);
      } else {
        jasmine.getEnv().addReporter(new jasmine.JUnitXmlReporter(newFolder, true, true));
      }
    });
  },
};

describe('login page', function() {

  var params = browser.params;

  it('should login successfully', function() {

});

"But, I want to use Protractor in a non-AngularJS app"

Sorry, you can't! :(

Just kidding, of course
you can! :)

You only need to access the webdriver instance by using browser.driver:


browser.driver.findElement(by.css('[data-ptor="submit-btn"]'));

It can be even more elegant

In your config.js

onPrepare: function() {
   global.dvr = browser.driver;
}

In your test

dvr.findElement(by.css('[data-ptor="submit-btn"]'));

Protractor waits for Angular to finish its work

Though you can tell it not to be that smart about your non-Angular app:

beforeEach(function() {
   return browser.ignoreSynchronization = true;
});

Let's make it more semantic?

In your config.js

onPrepare: function() {
   global.isAngularSite = function(flag) {
      browser.ignoreSynchronization = !flag;
   };
}

In your test

beforeEach(function() {
   isAngularSite(false); // isAngularSite(true), if it's an Angular app!
});
